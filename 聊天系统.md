### 服务端代码
~~~
import socket
import threading

# 服务器配置
HOST = "0.0.0.0"  # 监听所有网卡
PORT = 8888       # 自定义端口（1024以上）

def handle_client(client_socket, client_address):
    """处理单个客户端连接"""
    print(f"新连接：{client_address}")
    try:
        # 循环接收客户端消息
        while True:
            # 接收消息（最大1024字节）
            data = client_socket.recv(1024)
            if not data:
                break  # 客户端断开连接
            
            # 解码消息并打印
            message = data.decode("utf-8")
            print(f"【{client_address}】{message}")
            
            # 回复客户端
            reply = f"服务器已收到：{message}"
            client_socket.send(reply.encode("utf-8"))
    except Exception as e:
        print(f"连接异常 {client_address}：{e}")
    finally:
        # 关闭连接
        client_socket.close()
        print(f"连接关闭：{client_address}")

def start_server():
    """启动聊天服务器"""
    # 创建TCP socket
    server_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    # 允许端口复用（避免重启服务器时端口被占用）
    server_socket.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
    # 绑定地址和端口
    server_socket.bind((HOST, PORT))
    # 开始监听（最大5个等待连接）
    server_socket.listen(5)
    print(f"服务器启动成功，监听 {HOST}:{PORT}...")
    
    try:
        # 循环接受客户端连接
        while True:
            # 等待客户端连接（阻塞）
            client_socket, client_address = server_socket.accept()
            # 启动线程处理客户端（支持多客户端）
            client_thread = threading.Thread(target=handle_client, args=(client_socket, client_address))
            client_thread.daemon = True  # 主线程退出时子线程也退出
            client_thread.start()
    except KeyboardInterrupt:
        print("\n服务器正在关闭...")
    finally:
        server_socket.close()

if __name__ == "__main__":
    start_server()
~~~

#### 关于最后的多线程
~~~
client_thread = threading.Thread(target=handle_client, args=(client_socket, client_address))
~~~

| 参数                                     | 含义                     | 注意事项                           |
| -------------------------------------- | ---------------------- | ------------------------------ |
| `target=handle_client`                 | 线程执行的目标函数              | 只写函数名，**不加括号**（加括号会立即执行）       |
| `args=(client_socket, client_address)` | 传给 `handle_client` 的参数 | 必须是**元组**；单个参数也要加逗号：`(sock,)`  |
| `client_socket`                        | 与该客户端通信的专属套接字          | 由 `accept()` 返回，用于 `send/recv` |
| `client_address`                       | 客户端地址（IP + 端口）         | 用于日志 / 识别客户端                   |
- 如果不适用多线程会导致程序 [[串行]]与[[阻塞]]

```
import tkinter as tk
from tkinter import scrolledtext, messagebox
import socket
import threading

class NetworkChatApp:
    def __init__(self, root):
        self.root = root
        self.root.title("局域网聊天客户端")
        self.root.geometry("500x600")
        
        # 客户端socket（初始为空）
        self.client_socket = None
        self.connected = False
        
        # 创建UI
        self._create_widgets()
    
    def _create_widgets(self):
        """创建界面"""
        # 1. 连接服务器区域
        connect_frame = tk.Frame(self.root)
        connect_frame.pack(padx=10, pady=5, fill=tk.X)
        
        tk.Label(connect_frame, text="服务器IP：").pack(side=tk.LEFT)
        self.ip_entry = tk.Entry(connect_frame, width=15)
        self.ip_entry.insert(0, "127.0.0.1")  # 默认本地IP
        self.ip_entry.pack(side=tk.LEFT, padx=5)
        
        tk.Label(connect_frame, text="端口：").pack(side=tk.LEFT)
        self.port_entry = tk.Entry(connect_frame, width=8)
        self.port_entry.insert(0, "8888")
        self.port_entry.pack(side=tk.LEFT, padx=5)
        
        self.connect_btn = tk.Button(
            connect_frame,
            text="连接",
            bg="#4CAF50",
            fg="white",
            command=self.toggle_connect
        )
        self.connect_btn.pack(side=tk.LEFT, padx=5)
        
        # 2. 聊天记录区
        self.chat_area = scrolledtext.ScrolledText(
            self.root,
            wrap=tk.WORD,
            font=("微软雅黑", 12),
            state=tk.DISABLED
        )
        self.chat_area.pack(padx=10, pady=5, fill=tk.BOTH, expand=True)
        
        # 3. 输入区
        input_frame = tk.Frame(self.root)
        input_frame.pack(padx=10, pady=5, fill=tk.X)
        
        self.input_entry = tk.Entry(input_frame, font=("微软雅黑", 12))
        self.input_entry.pack(side=tk.LEFT, fill=tk.X, expand=True, padx=5)
        self.input_entry.bind("<Return>", self.send_message)
        
        self.send_btn = tk.Button(
            input_frame,
            text="发送",
            bg="#2196F3",
            fg="white",
            command=self.send_message,
            state=tk.DISABLED  # 未连接时禁用
        )
        self.send_btn.pack(side=tk.RIGHT, padx=5)
    
    def toggle_connect(self):
        """连接/断开服务器"""
        if not self.connected:
            # 连接服务器
            try:
                host = self.ip_entry.get()
                port = int(self.port_entry.get())
                
                # 创建socket并连接
                self.client_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
                self.client_socket.connect((host, port))
                self.connected = True
                
                # 更新UI
                self.connect_btn.config(text="断开", bg="#f44336")
                self.send_btn.config(state=tk.NORMAL)
                self._add_message("系统", "已成功连接服务器！", "system")
                
                # 启动线程接收消息
                recv_thread = threading.Thread(target=self.receive_messages)
                recv_thread.daemon = True
                recv_thread.start()
            except Exception as e:
                messagebox.showerror("错误", f"连接失败：{e}")
        else:
            # 断开连接
            self.connected = False
            if self.client_socket:
                self.client_socket.close()
                self.client_socket = None
            
            # 更新UI
            self.connect_btn.config(text="连接", bg="#4CAF50")
            self.send_btn.config(state=tk.DISABLED)
            self._add_message("系统", "已断开与服务器的连接！", "system")
    
    def receive_messages(self):
        """接收服务器消息（独立线程）"""
        while self.connected:
            try:
                data = self.client_socket.recv(1024)
                if not data:
                    break
                message = data.decode("utf-8")
                self._add_message("服务器", message, "server")
            except:
                break
    
    def send_message(self, event=None):
        """发送消息到服务器"""
        if not self.connected:
            messagebox.showwarning("提示", "请先连接服务器！")
            return
        
        msg = self.input_entry.get().strip()
        if not msg:
            return
        
        try:
            # 发送消息
            self.client_socket.send(msg.encode("utf-8"))
            self._add_message("我", msg, "user")
            self.input_entry.delete(0, tk.END)
        except Exception as e:
            messagebox.showerror("错误", f"发送失败：{e}")
            self.toggle_connect()  # 自动断开
    
    def _add_message(self, sender, content, tag):
        """添加消息到聊天区"""
        self.chat_area.config(state=tk.NORMAL)
        self.chat_area.insert(tk.END, f"【{sender}】\n{content}\n\n", tag)
        self.chat_area.config(state=tk.DISABLED)
        self.chat_area.see(tk.END)
        
        # 设置标签颜色
        self.chat_area.tag_config("system", foreground="#FF9800")  # 系统消息-橙色
        self.chat_area.tag_config("user", foreground="#2E8B57")    # 自己-绿色
        self.chat_area.tag_config("server", foreground="#1E90FF")  # 服务器-蓝色

if __name__ == "__main__":
    root = tk.Tk()
    app = NetworkChatApp(root)
    root.mainloop()
```

### 运行步骤

1. **启动服务器**：
    
    - 在一台电脑上运行`chat_server.py`，控制台显示`服务器启动成功，监听 0.0.0.0:8888...`
    - 查看该电脑的局域网 IP（Windows：`ipconfig`，Linux/Mac：`ifconfig`），比如`192.168.1.100`
    
2. **启动客户端**：
    
    - 在同一局域网的任意电脑（包括服务器本机）运行`chat_client.py`
    - 在客户端界面输入服务器 IP（如`192.168.1.100`）和端口`8888`，点击 “连接”
    - 连接成功后，输入消息点击 “发送”，即可和服务器通信
    

### 核心知识点

1. **Socket 编程**：`socket.socket()`创建 TCP 套接字，`bind()`绑定地址，`listen()`监听，`accept()`接受连接
2. **多线程**：服务器用`threading.Thread`处理多个客户端，客户端用独立线程接收消息（避免界面卡死）
3. **网络数据传输**：消息需用`encode()`编码为字节流发送，`decode()`解码接收
4. **GUI 状态管理**：根据连接状态启用 / 禁用按钮，更新界面提示
